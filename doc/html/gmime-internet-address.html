enum InternetAddressType

typedef enum {
	INTERNET_ADDRESS_NONE,
	INTERNET_ADDRESS_NAME,
	INTERNET_ADDRESS_GROUP
} InternetAddressType;

struct InternetAddress

struct InternetAddress {
	InternetAddressType type;
	gchar *name;
	union {
		gchar *addr;
		GList *members;
	} value;
};
