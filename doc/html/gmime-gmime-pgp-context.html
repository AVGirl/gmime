	GMIME_PGP_TYPE_PGP2,
	GMIME_PGP_TYPE_PGP5,
	GMIME_PGP_TYPE_PGP6,

g_mime_pgp_context_new ()

GMimePgpContext* g_mime_pgp_context_new     (GMimePgpType type,
                                             const char *path,
                                             GMimePgpPassphraseFunc get_passwd,
                                             gpointer user_data);

This creates a new GMimePgpContext object which is used to sign, verify, encrypt and decrypt streams.

type : One of GMIME_PGP_TYPE_[PGP2, PGP5, PGP6, GPG]
path : path to PGP binary
get_passwd : Password callback function
user_data : user data to pass to get_passwd
Returns : the new GMimePgpContext
