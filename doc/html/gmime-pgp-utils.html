enum PgpType

typedef enum {
	PGP_TYPE_NONE,
	PGP_TYPE_PGP2,
	PGP_TYPE_PGP5,
	PGP_TYPE_GPG
} PgpType;

enum PgpHashType

typedef enum {
	PGP_HASH_TYPE_NONE,
	PGP_HASH_TYPE_MD5,
	PGP_HASH_TYPE_SHA1
} PgpHashType;

pgp_decrypt ()

gchar*      pgp_decrypt                     (const gchar *ciphertext,
                                             gint cipherlen,
                                             gint *outlen,
                                             GMimeException *ex);

ciphertext : ciphertext to decrypt
cipherlen : ciphertext length
outlen : output length of the decrypted data (to be set by pgp_decrypt)
ex : exception
Returns :an allocated buffer containing the decrypted ciphertext. If the cleartext is plain text then you may treat it like a normal string as it will be NUL terminated, however outlen is also set in the case that the cleartext is a binary stream.
